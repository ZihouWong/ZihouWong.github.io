	void Insertion_Sort(double ar[], int n){
		int i, t, key;
		for(i = 1;i < n; i++){
			key = ar[i];
			t = i - 1;
			while (t >= 0 && ar[t]>key) {
				ar[t+1] = ar[t];
				t--;
			}
			ar[t+1] = key ;
		}	
	}
